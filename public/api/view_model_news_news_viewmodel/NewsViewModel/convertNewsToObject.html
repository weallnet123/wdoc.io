Object convertNewsToObject(News news) {
  final postObject = {
    'newsID': news.newsID ?? 0,
    'channelConfigID': news.channelConfigID,
    'languageID': news.languageID,
    'name': news.name,
    'title': news.title,
    'body': news.body,
    'newsURL': news.newsURL,
    'order': news.order ?? 0,
    'publishDate': news.publishDate != null
        ? news.publishDate!.toIso8601String()
        : DateTime.now().toIso8601String(),
    'enable': news.enable ?? 0,
    'openExternal': news.openExternal ?? 0,
    'articleUrl': news.articleUrl,
    'articleUrlSlug': news.articleUrlSlug,
    'articleDate':
        news.articleDate != null ? news.articleDate!.toIso8601String() : '',
    'articleImageUrl': news.articleImageUrl,
    'bodyImageResized': news.bodyImageResized,
    'hasVideo': news.hasVideo,
    'playVideoUrl': news.playVideoUrl,
    'code': news.code,
    'imageUrl': news.imageUrl,
    'totalShare': news.totalShare,
    'label': news.label,
    //end table database field
  };
  return postObject;
}
