Future trackingMedia(
    {required int? siteId,
    required String rand,
    required String mediaUrl,
    required int? mediaId,
    required String mediaType,
    required String? mediaTitle,
    required double? mediaLength,
    int mediaWatchedTime = 5,
    String namePlayer = 'Mobile Player'}) async {
  // ignore: unnecessary_this
  if (userCode == null) {
    userCode = anonymousDeviceId;
  }
  String MatomoAPI =
      '${MatomoConstants.DEV_MATOMO_HOST}/matomo.php?idsite=${siteId}&rec=1&uid=${userCode}&rand=${rand}&apiv=${apiVersion}';

  String trackMediaURLAPI =
      'ma_id=${mediaId}&ma_ti=${mediaTitle}&ma_re=${mediaUrl}&ma_mt=${mediaType}&ma_pn=${namePlayer}&ma_st=${mediaWatchedTime}&ma_le=${mediaLength.toString()}';

  String trackingUrl = '${MatomoAPI}&${trackMediaURLAPI}';
  DateTime now = DateTime.now();
  String formattedDate = DateFormat('yyyy-MM-dd HH:mm:ss').format(now);

  ViewRecord viewRecord = ViewRecord(
      mediaType: mediaType,
      mediaID: mediaId,
      viewDate: formattedDate,
      userId: userCode);

  final existedCalls = await checkDailyCalls(
    mediaId: mediaId,
    mediaType: mediaType,
    userID: userCode,
  );
  if (existedCalls >= 0 && mediaLength != null && mediaLength > 0) {
    try {
      var response = await _dioClient.get(trackingUrl);
      if (response != '') {
        if (existedCalls == 0) {
          //insert a new record
          _localDBHelper.insertViewRecord(viewRecord);
        } else {
          _localDBHelper.updateViewRecord(existedCalls, viewRecord);
        }
      }
    } catch (e) {}
  }
}
