static Route<dynamic> generateRoute(RouteSettings settings) {
  switch (settings.name) {
    case welcome:
      return MaterialPageRoute(
        settings: RouteSettings(name: welcome),
        builder: (context) => WelcomeScreen(),
      );
    case register:
      return MaterialPageRoute(
        settings: RouteSettings(name: register),
        builder: (context) => RegisterScreen(),
      );
    case media:
      return MaterialPageRoute(
        settings: RouteSettings(name: media),
        builder: (context) => TabWidget(),
      );
    case media_movie_detail:
      Map arr = settings.arguments as Map;
      return MaterialPageRoute(
        settings: RouteSettings(name: media_movie_detail),
        builder: (context) => MovieDetailScreen(
          movieID: arr['id'],
          code: arr['code'] ?? '',
        ),
      );
    case media_music_detail:
      Map arr = settings.arguments as Map;
      return MaterialPageRoute(
        settings: RouteSettings(name: media_music_detail),
        builder: (context) => MusicDetailScreen(
          musicID: arr['id'],
          code: arr['code'] ?? '',
        ),
      );
    case media_music_add_playlist:
      return MaterialPageRoute(
        settings: RouteSettings(name: media_music_add_playlist),
        builder: (context) => MusicAddPlaylistScreen(),
      );
    case media_music_create_playlist:
      return MaterialPageRoute(
        settings: RouteSettings(name: media_music_create_playlist),
        builder: (context) => MusicCreatePlaylistScreen(),
      );
    case media_video_detail:
      Map arr = settings.arguments as Map;
      return MaterialPageRoute(
        settings: RouteSettings(name: media_video_detail),
        builder: (context) => VideoDetailScreen(
          videoID: arr['id'],
          code: arr['code'] ?? '',
        ),
      );
    case channel:
      final channelCode = settings.arguments as String;
      return MaterialPageRoute(
        settings: RouteSettings(name: channel),
        builder: (context) => ChannelScreen(channelCode: channelCode),
      );
    case channelMovie:
      final channelCode = settings.arguments as String;
      return MaterialPageRoute(
        settings: RouteSettings(name: channelMovie),
        builder: (context) => ChannelMovies(channelCode: channelCode),
      );
    case channel_quicklook:
      final channelCode = settings.arguments as String;
      return MaterialPageRoute(
        settings: RouteSettings(name: channel_quicklook),
        builder: (context) => ChannelQuickLook(channelCode: channelCode),
      );
    case media_video_player:
      return MaterialPageRoute(
        settings: RouteSettings(name: media_video_player),
        builder: (context) => VideoPlayerScreen(),
      );
    case news:
      return MaterialPageRoute(
        settings: RouteSettings(name: news),
        builder: (context) => TabWidget(),
      );
    case news_detail:
      Map arr = settings.arguments as Map;
      return MaterialPageRoute(
        settings: RouteSettings(name: news_detail),
        builder: (context) => NewsDetailScreen(
          newsID: arr['id'],
          code: arr['code'] ?? '',
        ),
      );
    case news_web_view:
      {
        Map arr = settings.arguments as Map;
        return MaterialPageRoute(
          settings: RouteSettings(name: news_web_view),
          builder: (context) => NewsWebViewScreen(
            newsID: arr['id'],
            isEnable: arr['isEnable'],
            url: arr['url'] ?? '',
            code: arr['code'] ?? '',
          ),
        );
      }

    case news_create_category:
      return MaterialPageRoute(
        settings: RouteSettings(name: news_create_category),
        builder: (context) => NewsCreateCategoryScreen(),
      );
    case news_home:
      return MaterialPageRoute(
        settings: RouteSettings(name: news_home_all_networks),
        builder: (context) => NewsHomeScreen(),
      );

    case news_home_all_networks:
      return MaterialPageRoute(
        settings: RouteSettings(name: news_home_all_networks),
        builder: (context) => AllNetworksScreen(),
      );
    case home:
      return MaterialPageRoute(
        settings: RouteSettings(name: home),
        builder: (context) => TabWidget(),
      );
    case chat:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat),
        builder: (context) => TabWidget(),
      );
    case chat_recent:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_recent),
        builder: (context) => RecentListWidget(),
      );
    case chat_detail:
      final Map arr = settings.arguments as Map;
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_detail),
        builder: (context) => ChatDetailScreen(
          isFriendBlocker: arr['isFriendBlocker'] ?? false,
          isFriendBlocked: arr['isFriendBlocked'] ?? false,
          code: arr['code'] ?? '',
          chatBoxID: arr['chatBoxID'],
        ),
      );
    // final chatBoxCode = settings.arguments as String;
    // return MaterialPageRoute(
    //   settings: RouteSettings(name: chat_detail),
    //   builder: (context) => ChatDetailScreen(code: chatBoxCode),
    // );
    case chat_create:
      final isGroup = settings.arguments;
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_create),
        builder: (context) => ChatCreateScreen(isGroup: isGroup),
      );
    case chat_create_group:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_create_group),
        builder: (context) => CreateGroupChatScreen(),
      );
    case chat_more:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_more),
        builder: (context) => ChatMore(),
      );
    case chat_group_info:
      final chatBox = settings.arguments;
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_group_info),
        builder: (context) => GroupInfoScreen(chatBox: chatBox),
      );
    case chat_search_user:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_search_user),
        builder: (context) => ChatSearchUserScreen(),
      );
    case chat_search_contact:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_search_contact),
        builder: (context) => ChatSearchContactScreen(),
      );
    case chat_search_contact_group:
      return MaterialPageRoute(
        settings: RouteSettings(name: chat_search_contact_group),
        builder: (context) => ContactListForwardMessageWidget(),
      );
    case contact_screen:
      return MaterialPageRoute(
        settings: RouteSettings(name: contact_screen),
        builder: (context) => ContactScreen(),
      );
    case request_add_friend:
      return MaterialPageRoute(
          settings: RouteSettings(name: request_add_friend),
          builder: (context) => RequestScreen());

    case deals:
      return MaterialPageRoute(
        settings: RouteSettings(name: deals),
        builder: (context) => TabWidget(),
      );
    case favorite:
      final bool isBack = settings.arguments as bool;
      return MaterialPageRoute(
        settings: RouteSettings(name: favorite),
        builder: (context) => FavoriteScreen(isBack: isBack),
      );
    case events:
      return MaterialPageRoute(
        settings: RouteSettings(name: events),
        builder: (context) => EventsScreen(),
      );
    case events_detail:
      return MaterialPageRoute(
        settings: RouteSettings(name: events_detail),
        builder: (context) => EventsDetailScreen(),
      );
    case policy_register:
      return MaterialPageRoute(
        settings: RouteSettings(name: policy_register),
        builder: (context) => PolicyRegisterScreen(),
      );
    case delegate_register:
      return MaterialPageRoute(
        settings: RouteSettings(name: delegate_register),
        builder: (context) => DelegateRegisterScreen(),
      );
    case delegate_detail:
      return MaterialPageRoute(
        settings: RouteSettings(name: delegate_detail),
        builder: (context) => DelegateDetailScreen(),
      );
    case notification:
      //final bool isBack = settings.arguments as bool;
      return MaterialPageRoute(
        settings: RouteSettings(name: notification),
        builder: (context) => NotificationScreen(
            // isBack: isBack,
            ),
      );
    case menu:
      return MaterialPageRoute(
        settings: RouteSettings(name: menu),
        builder: (context) => MenuScreen(),
      );
    case more:
      return MaterialPageRoute(
        settings: RouteSettings(name: more),
        builder: (context) => MoreScreen(),
      );
    case more_media:
      final mediaParam = settings.arguments;
      return MaterialPageRoute(
        settings: RouteSettings(name: more_media),
        builder: (context) => MoreMediaScreen(mediaParam: mediaParam),
      );
    case review:
      return MaterialPageRoute(
        settings: RouteSettings(name: review),
        builder: (context) => ReviewScreen(),
      );
    case feedback:
      return MaterialPageRoute(
        settings: RouteSettings(name: feedback),
        builder: (context) => FeedBackScreen(),
      );
    case profile:
      return MaterialPageRoute(
        settings: RouteSettings(name: profile),
        builder: (context) => ProfileScreen(),
      );
    case profileDetail:
      return MaterialPageRoute(
        settings: RouteSettings(name: profileDetail),
        builder: (context) => ProfileDetailScreen(),
      );
    case change_password:
      return MaterialPageRoute(
        settings: RouteSettings(name: change_password),
        builder: (context) => ChangePasswordScreen(),
      );
    case block_user:
      return MaterialPageRoute(
        settings: RouteSettings(name: block_user),
        builder: (context) => BlockUserScreen(),
      );
    case influencer:
      final int influencerID = settings.arguments as int;
      return MaterialPageRoute(
        settings: RouteSettings(name: influencer),
        builder: (context) => InfluencerScreen(influencerID: influencerID),
      );
    case watch_wf:
      final WatchWFParam param = settings.arguments as WatchWFParam;
      return MaterialPageRoute(
        settings: RouteSettings(name: watch_wf),
        builder: (context) => WatchWFScreen(param: param),
      );
    case search_all:
      return MaterialPageRoute(
        settings: RouteSettings(name: search_all),
        builder: (context) => SearchScreen(),
      );
    default:
      return MaterialPageRoute(
        builder: (_) => Scaffold(
          body: Center(child: Text('No route defined for ${settings.name}')),
        ),
      );
  }
}
