Object convertProductToObject(Product product) {
  final postObject = {
    'productID': product.productID ?? 0,
    'channelConfigID': product.channelConfigID ?? 0,
    'productBrandID': product.productBrandID,
    'currencyID': product.currencyID,
    'productSubCategoryID': product.productSubCategoryID,
    'productSupplyID': product.productSupplyID,
    'price': product.price ?? 0,
    'name': product.name,
    'code': product.code,
    'productCode': product.productCode,
    'imageURL': product.imageURL,
    'videoURL': product.videoURL,
    'quantity': product.quantity ?? 0,
    'expirationDate': product.expirationDate != null
        ? product.expirationDate!.toIso8601String()
        : '',
    'discountPercent': product.discountPercent,
    'discountValue': product.discountValue,
    'enable': product.enable ?? 0,
    'publishDate': product.publishDate != null
        ? product.publishDate!.toIso8601String()
        : '',
    //end table database field
  };
  return postObject;
}
