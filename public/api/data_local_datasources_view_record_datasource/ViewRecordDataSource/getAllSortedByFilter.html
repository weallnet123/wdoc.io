Future<List<ViewRecord>> getAllSortedByFilter({List<Filter>? filters}) async {
  //creating finder
  final finder = Finder(
      filter: filters != null ? Filter.and(filters) : null,
      sortOrders: [SortOrder(DBConstants.FIELD_ID)]);

  final recordSnapshots = await _viewRecordStore.find(_db, finder: finder);

  // Making a List<Country> out of List<RecordSnapshot>
  return recordSnapshots.map((snapshot) {
    final viewRecord = ViewRecord.fromJson(snapshot.value);
    // An ID is a key of a record from the database.
    return viewRecord;
  }).toList();
}
