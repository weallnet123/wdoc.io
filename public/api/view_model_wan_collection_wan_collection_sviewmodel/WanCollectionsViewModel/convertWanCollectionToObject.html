Object convertWanCollectionToObject(WanCollection wanCollection) {
  final postObject = {
    'wanCollectionID': wanCollection.wanCollectionID ?? 0,
    'code': wanCollection.code,
    'name': wanCollection.name,
    'nameEn': wanCollection.nameEn,
    'type': wanCollection.type,
    'itemType': wanCollection.itemType,
    'searchKeyword': wanCollection.searchKeyword,
    'publishDate': wanCollection.publishDate != null
        ? wanCollection.publishDate!.toIso8601String()
        : '',
    'description': wanCollection.description,
    'enable': wanCollection.enable ?? 0,
    'order': wanCollection.order ?? 0,
    //end table database field
  };
  return postObject;
}
