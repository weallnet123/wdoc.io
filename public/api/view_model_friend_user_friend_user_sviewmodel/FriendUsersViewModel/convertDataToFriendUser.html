FriendUser convertDataToFriendUser(Map<String, dynamic> json) {
  FriendUser item = FriendUser.fromJson(json);
  if (item.friendUserProfileImageUrl != null &&
      item.friendUserProfileImageUrl!.length > 30) {
    String temp = item.friendUserProfileImageUrl!;
    final index = temp.lastIndexOf('.');
    item.sFriendUserProfileImageUrl =
        temp.replaceFirst(RegExp('.'), '_small.', index);
    item.mFriendUserProfileImageUrl =
        temp.replaceFirst(RegExp('.'), '_medium.', index);
  }
  return item;
}
