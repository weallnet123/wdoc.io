Future<CountryList> getCountrysFromDb() async {
  // Country list
  var arrCountry;

  // fetching data
  final recordSnapshots = await _countrysStore.find(_db);

  // Making a List<Country> out of List<RecordSnapshot>
  if (recordSnapshots.length > 0) {
    arrCountry = CountryList(
        arrCountry: recordSnapshots.map((snapshot) {
      final country = Country.fromJson(snapshot.value);
      // An ID is a key of a record from the database.
      country.countryID = snapshot.key;
      return country;
    }).toList());
  }

  return arrCountry;
}
