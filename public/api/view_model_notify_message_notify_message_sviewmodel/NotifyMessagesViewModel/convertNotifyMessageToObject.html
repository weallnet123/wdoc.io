Object convertNotifyMessageToObject(NotifyMessage notifyMessage) {
  final postObject = {
    'notifyMessageID': notifyMessage.notifyMessageID ?? 0,
    'notifyToUserID': notifyMessage.notifyToUserID,
    'notifyUserIDSender': notifyMessage.notifyUserIDSender,
    'message': notifyMessage.message,
    'imageUserSender': notifyMessage.imageUserSender,
    'actionLink': notifyMessage.actionLink,
    'messageType': notifyMessage.messageType,
    'status': notifyMessage.status,
    'notifyDate': notifyMessage.notifyDate != null
        ? notifyMessage.notifyDate!.toIso8601String()
        : '',
    'notifyDateM': notifyMessage.notifyDateM,
    'actionDate': notifyMessage.actionDate != null
        ? notifyMessage.actionDate!.toIso8601String()
        : '',
    'title': notifyMessage.title,
    'subtitle': notifyMessage.subtitle,
    'isSeen': notifyMessage.isSeen ?? false,
    //end table database field
  };
  return postObject;
}
