Future delete(Influencer influencer) async {
  apiStatusStore = ApiStatusStore();
  processing.setProcessing('Influencers', true, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['Influencers']);
  try {
    String searchUrl = getUrls['remove'].toString();
    final res =
        await _dioClient.delete('$searchUrl?id=${influencer.influencerID}');
    if (res['result'] == 'Success') {
      arrInfluencer.removeWhere(
          (item) => item.influencerID == influencer.influencerID);
      paging.resetPaging(
        paging.pageIndex,
        paging.pageSize,
        paging.totalCount - 1,
      );
      apiStatusStore.setApiSuccess();
    } else {
      apiStatusStore.setApiUnsuccess(
        res['errorMessage'].toString(),
        res['errorMessage'].toString(),
      );
      Catcher.reportCheckedError(
          res['errorMessage'].toString(), StackTrace.empty);
    }
  } catch (error, stackTrace) {
    NetworkExceptionText errorMessage =
        NetworkException.getDioException(error);
    apiStatusStore.setApiError(
      errorMessage.errorMessageVI,
      errorMessage.errorMessageUS,
    );
    Catcher.reportCheckedError(error, stackTrace);
  }
  processing.setProcessing('Influencers', false, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['Influencers']);
}
