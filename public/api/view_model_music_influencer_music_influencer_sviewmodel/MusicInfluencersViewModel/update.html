Future update(Map<String, dynamic> params) async {
  apiStatusStore = ApiStatusStore();
  processing.setProcessing('MusicInfluencers', true, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['MusicInfluencers']);
  try {
    String searchUrl = getUrls['saveParam'].toString();
    final res = await _dioClient.post(searchUrl,
        data: jsonEncode(_collectUpdateParam(params)));
    if (res['result'] == 'Success') {
      MusicInfluencer musicInfluencer =
          convertDataToMusicInfluencer(res['musicInfluencer']);
      int index = arrMusicInfluencer.indexWhere((item) =>
          item.musicInfluencerID == musicInfluencer.musicInfluencerID);
      if (index != -1) {
        arrMusicInfluencer[index] = musicInfluencer;
      }
      apiStatusStore.setApiSuccess();
    } else {
      apiStatusStore.setApiUnsuccess(
        res['errorMessage'].toString(),
        res['errorMessage'].toString(),
      );
      Catcher.reportCheckedError(
          res['errorMessage'].toString(), StackTrace.empty);
    }
  } catch (error, stackTrace) {
    NetworkExceptionText errorMessage =
        NetworkException.getDioException(error);
    apiStatusStore.setApiError(
      errorMessage.errorMessageVI,
      errorMessage.errorMessageUS,
    );
    Catcher.reportCheckedError(error, stackTrace);
  }
  processing.setProcessing('MusicInfluencers', false, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['MusicInfluencers']);
}
