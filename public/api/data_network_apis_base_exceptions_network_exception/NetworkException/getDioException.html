static NetworkExceptionText getDioException(error) {
  NetworkExceptionText errorMessage = NetworkExceptionText(
    errorMessageUS: '',
    errorMessageVI: '',
  );
  if (error is Exception) {
    try {
      if (error is DioError) {
        switch (error.type) {
          case DioErrorType.cancel:
            errorMessage.errorMessageUS = 'Request Cancelled';
            errorMessage.errorMessageVI = 'Yêu cầu bị hủy';
            break;
          case DioErrorType.connectTimeout:
            errorMessage.errorMessageUS = 'Connection request timeout';
            errorMessage.errorMessageVI = 'Yêu cầu kết nối hết hạn';
            break;
          case DioErrorType.other:
            errorMessage.errorMessageUS = 'No internet connection';
            errorMessage.errorMessageVI = 'Không có kết nối mạng';
            break;
          case DioErrorType.receiveTimeout:
            errorMessage.errorMessageUS =
                'Send timeout in connection with API server';
            errorMessage.errorMessageVI = 'Hết hạn kết nối với máy chủ';
            break;
          case DioErrorType.response:
            switch (error.response?.statusCode) {
              case 400:
                errorMessage.errorMessageUS = 'Unauthorised request';
                errorMessage.errorMessageVI =
                    'Không được phép thực hiện yêu cầu';
                break;
              case 401:
                errorMessage.errorMessageUS = 'Unauthorised request';
                errorMessage.errorMessageVI =
                    'Không được phép thực hiện yêu cầu';
                break;
              case 403:
                errorMessage.errorMessageUS = 'Unauthorised request';
                errorMessage.errorMessageVI =
                    'Không được phép thực hiện yêu cầu';
                break;
              case 404:
                errorMessage.errorMessageUS = 'Not found';
                errorMessage.errorMessageVI = 'Không tìm thấy';
                break;
              case 409:
                errorMessage.errorMessageUS = 'Error due to a conflict';
                errorMessage.errorMessageVI = 'Lỗi bởi vì sự xung đột';
                break;
              case 408:
                errorMessage.errorMessageUS = 'Connection request timeout';
                errorMessage.errorMessageVI = 'Yêu cầu kết nối hết hạn';
                break;
              case 500:
                errorMessage.errorMessageUS = 'Internal Server Error';
                errorMessage.errorMessageVI = 'Lỗi nội bộ';
                break;
              case 503:
                errorMessage.errorMessageUS = 'Service unavailable';
                errorMessage.errorMessageVI = 'Dịch vụ không phù hợp';
                break;
              default:
                var responseCode = error.response?.statusCode;
                errorMessage.errorMessageUS =
                    'Received invalid status code: $responseCode';
                errorMessage.errorMessageVI =
                    'Đã nhận mã trạng thái không hợp lệ: $responseCode';
            }
            break;
          case DioErrorType.sendTimeout:
            errorMessage.errorMessageUS =
                'Send timeout in connection with API server';
            errorMessage.errorMessageVI = 'Hết hạn kết nối với máy chủ';
            break;
        }
      } else if (error is SocketException) {
        errorMessage.errorMessageUS = 'No internet connection';
        errorMessage.errorMessageVI = 'Không có kết nối mạng';
      } else {
        errorMessage.errorMessageUS = 'Unexpected error occurred';
        errorMessage.errorMessageVI = 'Đã xảy ra lỗi không mong muốn';
      }
      return errorMessage;
    } on FormatException catch (_) {
      errorMessage.errorMessageUS = 'Unexpected error occurred';
      errorMessage.errorMessageVI = 'Đã xảy ra lỗi không mong muốn';
      return errorMessage;
    } catch (_) {
      errorMessage.errorMessageUS = 'Unexpected error occurred';
      errorMessage.errorMessageVI = 'Đã xảy ra lỗi không mong muốn';
      return errorMessage;
    }
  } else {
    if (error.toString().contains('is not a subtype of')) {
      errorMessage.errorMessageUS = 'Unable to process the data';
      errorMessage.errorMessageVI = 'Không thể xử lý dữ liệu';
      return errorMessage;
    } else {
      errorMessage.errorMessageUS = 'Unexpected error occurred';
      errorMessage.errorMessageVI = 'Đã xảy ra lỗi không mong muốn';
      return errorMessage;
    }
  }
}
