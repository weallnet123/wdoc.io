String totalConverter(int value) {
  int x = 0, y = 0;
  List<String> ls = ['K', 'M'];
  if (value >= 1000 && value < 1000000) {
    x = value ~/ 1000;
    y = value ~/ 100 - x * 10;
    if (y == 0) {
      return '${x}${ls[0]}';
    }
    return '${x}.${y}${ls[0]}';
  } else if (value >= 1000000 && value < 1000000000) {
    x = value ~/ 1000000;
    y = value ~/ 100000 - x * 10;
    if (y == 0) {
      return '${x}${ls[1]}';
    }
    return '${x}.${y}${ls[1]}';
  }
  return '${value}';
}
