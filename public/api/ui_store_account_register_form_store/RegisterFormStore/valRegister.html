        if (!validCharacters.hasMatch(val)) {

        msg[Languages.langVI] = 'Mật khẩu phải có ít nhất 6 ký tự';
      } else {
        final validCharacters = RegExp(r'(?=.*\d)(?=.*[a-z])(?=.*[A-Z])');
        if (!validCharacters.hasMatch(val)) {
          msg[Languages.langEN] =
              'The password must contain at least 1 number, 1 \nlowercase character and 1 uppercase character';
          msg[Languages.langVI] =
              'Mật khẩu phải chứa ít nhất 1 chữ số, 1 kí tự viết thường và \n1 kí tự viết hoa';
        }
      }
      break;
    case 'confirmPassword':
      if (val == null || val.isEmpty) {
        msg[Languages.langEN] = '*';
        msg[Languages.langVI] = '*';
      } else if (val != model.password) {
        msg[Languages.langEN] = 'Password does not match';
        msg[Languages.langVI] = 'Không khớp với mật khẩu';
      }
      break;
  }
  return msg;
