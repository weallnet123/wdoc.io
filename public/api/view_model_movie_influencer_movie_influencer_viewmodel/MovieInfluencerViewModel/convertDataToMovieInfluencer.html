MovieInfluencer convertDataToMovieInfluencer(Map<String, dynamic> json) {
  MovieInfluencer item = MovieInfluencer.fromJson(json);
  if(item.influencerImageUrl != null && item.influencerImageUrl!.length > 30){
    String temp = item.influencerImageUrl!;
    final index = temp.lastIndexOf('.');
    item.sInfluencerImageUrl = temp.replaceFirst(RegExp('.'), '_small.', index);
    item.mInfluencerImageUrl = temp.replaceFirst(RegExp('.'), '_medium.', index);
  }
  return item;
}
