Describes the part of the user interface represented by this widget.

The framework calls this method when this widget is inserted into the tree in a given BuildContext and when the dependencies of this widget change (e.g., an InheritedWidget referenced by this widget changes). This method can potentially be called in every frame and should not have any side effects beyond building a widget.

The framework replaces the subtree below this widget with the widget returned by this method, either by updating the existing subtree or by removing the subtree and inflating a new subtree, depending on whether the widget returned by this method can update the root of the existing subtree, as determined by calling Widget.canUpdate.

Typically implementations return a newly created constellation of widgets that are configured with information from this widget's constructor and from the given BuildContext.

The given BuildContext contains information about the location in the tree at which this widget is being built. For example, the context provides the set of inherited widgets for this location in the tree. A given widget might be built with multiple different BuildContext arguments over time if the widget is moved around the tree or if the widget is inserted into the tree in multiple places at once.

The implementation of this method must only depend on:

  • the fields of the widget, which themselves must not change over time, and
  • any ambient state obtained from the context using BuildContext.dependOnInheritedWidgetOfExactType.

If a widget's build method is to depend on anything else, use a StatefulWidget instead.

See also:

  • StatelessWidget, which contains the discussion on performance considerations.

@override
Widget build(BuildContext context) {
  String? avatar = chatBox.sAvatar;
  String? rAvatar = chatBox.avatar;
  String? title = chatBox.name;
  if (chatBox.chatBoxUsers != null && chatBox.chatBoxUsers!.length == 2) {
    ChatBoxUser? user = chatBox.chatBoxUsers!.firstWhereOrNull(
      (u) => u.userID != ownerID,
    );
    if (user != null) {
      avatar = user.sUserImageUrl;
      rAvatar = user.userImageUrl;
      title = user.userFullName;
    }
  }
  return GestureDetector(
    behavior: HitTestBehavior.opaque,
    onTap: () => onTap(chatBox),
    onLongPress: () async {
      if (chatBox.isPublic ?? false) {
        appStore.initBottomSheet.setIsShowBottomSheet(true);
        return await showModalBottomSheet(
          context: context,
          backgroundColor: Colors.transparent,
          builder: (BuildContext context) {
            return BottomModalSheetWidget(
              items: [
                BottomModalSheetItemWidget(
                  text: AppLocalizations.of(context)
                      .translate('chat_label_info'),
                  onTap: () {
                    Navigator.popAndPushNamed(
                      context,
                      Routes.chat_group_info,
                      arguments: chatBox,
                    );
                  },
                ),
                BottomModalSheetItemWidget(
                  text: AppLocalizations.of(context)
                      .translate('chat_label_out_group'),
                  textColor: Colors.red,
                  onTap: () {
                    onOutGroup(chatBox);
                    Navigator.pop(context);
                  },
                ),
                BottomModalSheetItemWidget(
                  text:
                      AppLocalizations.of(context).translate('button_close'),
                  onTap: () {
                    Navigator.pop(context);
                  },
                ),
              ],
            );
          },
        ).whenComplete(() {
          appStore.initBottomSheet.closeBottomSheet();
        });
      } else {
        appStore.initBottomSheet.setIsShowBottomSheet(true);
        return await showModalBottomSheet(
                context: context,
                backgroundColor: Colors.transparent,
                builder: (BuildContext context) {
                  return BottomModalSheetWidget(
                    items: [
                      if (!chatBox.isFriendBlocked!) ...[
                        BottomModalSheetItemWidget(
                          text: AppLocalizations.of(context)
                              .translate('common_label_block'),
                          onTap: () {
                            onBlockChatBox(chatBox);
                          },
                        )
                      ],
                      //  BottomModalSheetItemWidget(
                      //     text: AppLocalizations.of(context)
                      //         .translate('button_delete'),
                      //     onTap: () {
                      //       onCancelChatBox(chatBox);
                      //       Navigator.pop(context);
                      //     },
                      //   ),
                      BottomModalSheetItemWidget(
                        text: AppLocalizations.of(context)
                            .translate('button_close'),
                        onTap: () {
                          Navigator.pop(context);
                        },
                      ),
                    ],
                  );
                })
            .whenComplete(() => appStore.initBottomSheet.closeBottomSheet());
      }
    },
    child: Padding(
      padding: const EdgeInsets.only(
          top: 8.0, bottom: 8.0, left: 16.0, right: 16.0),
      child: Row(
        children: [
          Container(
            height: Dimens.DEFAULT_SIZE_AVATAR,
            child: _Avatar(
              imageUrl: avatar,
              rImageUrl: rAvatar,
            ),
          ),
          SizedBox(width: 8.0),
          Expanded(
            child: Container(
              height: Dimens.DEFAULT_SIZE_AVATAR,
              // padding: EdgeInsets.only(top: 0.0),
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  Row(
                    children: [
                      _Title(name: '$title'),
                      SizedBox(width: 8.0),
                      // _Timer(timer: '20 min'),
                    ],
                  ),
                  // SizedBox(height: 8.0),
                  _Message(
                    isMySender: appStore.account.userId ==
                        latestChatBoxMessage.senderID,
                    messageType: latestChatBoxMessage.messageType ?? 'null',
                    userSenderName: latestChatBoxMessage.senderFullName,
                    message: latestChatBoxMessage.tagUsers != null
                        ? _formatTagText(latestChatBoxMessage.content!)
                        : latestChatBoxMessage.content != null
                            ? latestChatBoxMessage.content!
                            : '',
                  ),
                ],
              ),
            ),
          ),
          latestChatBoxMessage.postedDateM != null
              ? _Timer(
                  timer: DateFormat.Md().format(
                      DateTime.fromMicrosecondsSinceEpoch(
                          latestChatBoxMessage.postedDateM!.toInt() * 1000)))
              : SizedBox.shrink(),
        ],
      ),
    ),
  );
}
