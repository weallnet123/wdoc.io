Future initVideoScreenshot() async {
  apiStatusStore = ApiStatusStore();
  processing.setProcessing('VideoScreenshot', true, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['VideoScreenshot']);
  try {
    if (arrSearchParam.keys.length > 0) {
      String getVideoScreenshot = getUrls['getParam'].toString();
      final res = await _dioClient.post('$getVideoScreenshot',
          data: jsonEncode(_collectGetParams()));
      if (res['result'] == 'Success') {
        videoScreenshot =
            convertDataToVideoScreenshot(res['videoScreenshot']);
        apiStatusStore.setApiSuccess();
      } else {
        apiStatusStore.setApiUnsuccess(
          res['errorMessage'].toString(),
          res['errorMessage'].toString(),
        );
        Catcher.reportCheckedError(
            res['errorMessage'].toString(), StackTrace.empty);
      }
    } else {
      if (videoScreenshotID! > 0) {
        String getVideoScreenshot = getUrls['getVideoScreenshot'].toString();
        final res =
            await _dioClient.get('$getVideoScreenshot?id=$videoScreenshotID');
        if (res['result'] == 'Success') {
          videoScreenshot =
              convertDataToVideoScreenshot(res['videoScreenshot']);
          apiStatusStore.setApiSuccess();
        } else {
          apiStatusStore.setApiUnsuccess(
            res['errorMessage'].toString(),
            res['errorMessage'].toString(),
          );
          Catcher.reportCheckedError(
              res['errorMessage'].toString(), StackTrace.empty);
        }
      } else {
        String getVideoScreenshot = getUrls['getInit'].toString();
        final res = await _dioClient.get('$getVideoScreenshot');
        if (res['result'] == 'Success') {
          videoScreenshot =
              convertDataToVideoScreenshot(res['videoScreenshot']);
          apiStatusStore.setApiSuccess();
        } else {
          apiStatusStore.setApiUnsuccess(
            res['errorMessage'].toString(),
            res['errorMessage'].toString(),
          );
          Catcher.reportCheckedError(
              res['errorMessage'].toString(), StackTrace.empty);
        }
      }
    }
  } catch (error, stackTrace) {
    NetworkExceptionText errorMessage =
        NetworkException.getDioException(error);
    apiStatusStore.setApiError(
      errorMessage.errorMessageVI,
      errorMessage.errorMessageUS,
    );
    Catcher.reportCheckedError(error, stackTrace);
  }
  processing.setProcessing('VideoScreenshot', false, '');
  apiStatusStore.setApiLoading(processing.arrProcessing['VideoScreenshot']);
}
